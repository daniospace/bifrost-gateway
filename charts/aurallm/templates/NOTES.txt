Get the application URL:
{{- if .Values.ingress.enabled }}
{{- range .Values.ingress.hosts }}
  http{{ if $.Values.ingress.tls }}s{{ end }}://{{ .host }}
{{- end }}
{{- else if eq .Values.service.type "NodePort" }}
  export NODE_PORT=$(kubectl get svc {{ include "aurallm.fullname" . }} -o jsonpath="{.spec.ports[0].nodePort}")
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo "http://$NODE_IP:$NODE_PORT"
{{- else }}
  kubectl port-forward svc/{{ include "aurallm.fullname" . }} 8085:8085
  echo "http://localhost:8085"
{{- end }}

To inject provider API keys:
  kubectl create secret generic aurallm-secrets \
    --from-literal=openai-api-key=sk-... \
    --from-literal=anthropic-api-key=sk-ant-...

Then reference it in values.yaml:
  extraEnv:
    - name: OPENAI_API_KEY
      valueFrom:
        secretKeyRef:
          name: aurallm-secrets
          key: openai-api-key
